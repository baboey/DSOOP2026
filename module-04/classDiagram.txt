classDiagram
    class App {
        +main()
    }

    class Human {
        -String name
        -HumanDesire hDesire
        -PrimalDesire pDesire
        ~Human(String N)
        +setName(String N)
        +getName() String
        +setHumanDesire(String title, int morality)
        +setPrimalDesire(String title, String necessity)
        +printHumanDesire()
        +printPrimalDesire()
    }

    class Desire {
        <<abstract>>
        #String title
        #String kind
        ~Desire()
        ~Desire(String T, String K)
        +getTitle() String
        +getKind() String
    }

    class PrimalDesire {
        -String necessity
        ~PrimalDesire()
        ~PrimalDesire(String T, String N)
        +set(String T, String N)
        +getNecessity() String
    }

    class HumanDesire {
        -int morality
        ~HumanDesire()
        ~HumanDesire(String T, int M)
        +set(String T, int M)
        +getMorality() int
    }

    %% Relationships
    App ..> Human : instantiates
    Human *-- HumanDesire : composes
    Human *-- PrimalDesire : composes
    PrimalDesire --|> Desire : extends
    HumanDesire --|> Desire : extends
